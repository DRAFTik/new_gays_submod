chinese_reunification = { 
	
	JAP_intervention_timer = { #of communist MAN
		icon = generic_prepare_civil_war

		allowed = { always = no } #activated after surrender of pro-jap MAN
		
		available = { always = no }
		
		cancel_trigger = {
			OR = {
				is_subject = yes
				has_capitulated = yes
				has_country_flag = JAP_will_surrender_MGM
				has_global_flag = JAP_will_surrender
			}
		}

		is_good = no
		days_mission_timeout = 30

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): cancel decision JAP_intervention_timer"
		}

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): timeout decision JAP_intervention_timer"
			if = {
				limit = { tag = JAP }
				every_country = {
					limit = { has_country_flag = JAP_intervention_flag } 
					if = {
						limit = { is_subject = no has_capitulated = no }
						JAP = {
							declare_war_on = {
								target = PREV
								type = annex_everything
							}
						}
					}
					clr_country_flag = JAP_intervention_flag
				}
			}
			else = {
				effect_tooltip = {
					every_country = {
						limit = { has_country_flag = JAP_intervention_flag } 
						JAP = {
							declare_war_on = {
								target = PREV
								type = annex_everything
							}
						}
					}
				}
			}
		}
	}
	
	JAP_capitulation_timer_MGM = { #?	from KR	
		icon = revolt

		allowed = { always = no } #activated by manchukou.22

		cancel_trigger = {
			OR = {
				NOT = { has_war_with = MGM } #has_war
				is_subject = yes
				has_capitulated = yes
				#has_active_mission = JAP_capitulation_timer
			}
		}

		is_good = no
		days_mission_timeout = JAP_capitulation_timeout?90
		selectable_mission = yes

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): cancel decision JAP_capitulation_timer_MGM"
		}
		
		ai_will_do = {
			factor = 0
			modifier = {
				add = 1000
				OR = {
					has_country_flag = JAP_will_surrender_MGM
					has_global_flag = JAP_will_surrender
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): complete decision JAP_capitulation_timer_MGM"
			set_country_flag = JAP_will_surrender_MGM
			add_stability = -0.05
			add_war_support = -0.07
			add_political_power = -15
			every_country = {
				limit = { original_tag = MAN NOT = { tag = MAN } }
				country_event = manchukou.67 
			} 
		}

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): timeout decision JAP_capitulation_timer_MGM"
			add_stability = -0.03
			add_war_support = -0.05
			add_political_power = -10
			if = {
				limit = { has_idea = JAP_protracted_military_campaign }
				swap_ideas = {
					remove_idea = JAP_protracted_military_campaign
					add_idea = JAP_protracted_military_campaign_1
				}
			}
			else_if = {
				limit = { has_idea = JAP_protracted_military_campaign_1 }
				swap_ideas = {
					remove_idea = JAP_protracted_military_campaign_1
					add_idea = JAP_protracted_military_campaign_2
				}
			}
			else_if = {
				limit = { has_idea = JAP_protracted_military_campaign_2 }
				swap_ideas = {
					remove_idea = JAP_protracted_military_campaign_2
					add_idea = JAP_protracted_military_campaign_3
				}
				set_country_flag = JAP_will_surrender_MGM
			}
			else_if = {
				limit = { NOT = { has_idea = JAP_protracted_military_campaign_3 } }
				add_ideas = JAP_protracted_military_campaign
			}
			set_variable = { JAP_capitulation_timeout = 30 }
			activate_mission = JAP_capitulation_timer_MGM
		}
	}
}
