	option = {
		name = hebei.5.a
			if = {
				limit = {
					check_variable = { EHB_japan_influence_total < 13 }
				}
				country_event = {
					id = hebei.6
					days = 10
				}
			}
			if = {
				limit = {
					check_variable = { EHB_japan_influence_total > 12 }
					check_variable = { EHB_japan_influence_total < 21 }
				}
				country_event = {
					id = hebei.7
					days = 10
				}
			}

				limit = {

				}

					id = hebei.8

				}
			}
		}
