	id = syria.50
	title = syria.50.t
	desc = syria.50.desc
	picture = GFX_report_event_soviet_invasion_map
	is_triggered_only = yes
	
	option = {
		name = syria.50.a
		ai_chance = {
			factor = 1
		}
		FROM = { country_event = { id = syria.51 days = 1 } }
	}
	option = {
		name = syria.50.b
		ai_chance = {
			factor = 99
			modifier = {
				factor = 0
				OR = {
					has_government = democratic
					strength_ratio = {
						tag = FROM
						ratio < 1
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					controls_state = 995
				}
			}
			FROM = {
				transfer_state = 413
			}
		}
		else = {
			release = KHU
			FROM = {
				set_autonomy = {
					target = KHU
					autonomy_state = autonomy_integrated_puppet
				}
			}
		}
	}
}
